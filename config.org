* Fonts
#+BEGIN_SRC emacs-lisp
  (defun dejavu-font-check (&optional frame)
    (when frame
      (select-frame frame))
    (condition-case nil
        (set-frame-font
         "TerminessTTF Nerd Font Mono:style=Medium")
      (error
       (set-frame-font
        "DejaVu Sans Mono"))))
  (dejavu-font-check)
  (set-face-attribute 'default nil :height 120)
  (add-hook 'after-make-frame-functions 'dejavu-font-check)
